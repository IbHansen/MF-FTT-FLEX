@IDENTITY SIGMA_OIL_OIL  = SQR(SIGMA_OIL^2 * SIGMA_OIL^2)

@IDENTITY SIGMA_OIL_HYDRO  = SQR(SIGMA_OIL^2 * SIGMA_HYDRO^2)
